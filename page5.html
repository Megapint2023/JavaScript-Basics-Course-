<script>
 function greeting() {
   var getHours = new Date().getHours(); 
   var greeting;
   var color;
   var fontWeight;

   if (getHours >= 5 && getHours < 10) {
       greeting = "STATUS: AFK";
       color = "purple";
       fontWeight = "bold";
   } 
  else if (getHours >= 10 && getHours < 14) {
           greeting = "10:00 - 14:00 -> My day in progress. I am probably coding." + "<br>" +
           "STATUS: Online.";
           color = "green";
           fontWeight = "bold";  
  }
  else if (getHours >= 14 && getHours < 17) {
           greeting = "14:00 - 17:00 -> My day in progress. I am probably coding." + "<br>" +
           "STATUS: Online.";
           color = "green";
           fontWeight = "bold";
  }
  else if (getHours >= 17 && getHours < 23) {
           greeting = "17:00 - 23:00 -> Dont not disturb, unless its important." + "<br>" +
           "STATUS: DND.";  
           color = "red";
           fontWeight = "bold";
  }
   else {
       greeting = "Brain status -> Recharging." + "<br>" +
           "STATUS: AFK";
       color = "purple";
       fontWeight = "bold";
   }
   document.getElementById("greeting").innerHTML = greeting;
   document.getElementById("greeting").style.color = color;
 }
 </script>
