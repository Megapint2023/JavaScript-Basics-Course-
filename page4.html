function throw_dice(min, max) {
 var noppa = Math.floor(Math.random() * (max - min + 1) ) + min;


if (noppa == 1) {
    document.getElementById("random_dice_image").src = "images/dice1.jpg";
  } else if (noppa == 2) {
    document.getElementById("random_dice_image").src = "images/dice2.jpg";
  } else if (noppa == 3) {
    document.getElementById("random_dice_image").src = "images/dice3.jpg";
  } else if (noppa == 4) {
    document.getElementById("random_dice_image").src = "images/dice4.jpg";
  } else if (noppa == 5) {
    document.getElementById("random_dice_image").src = "images/dice5.jpg";
  } else {
    document.getElementById("random_dice_image").src = "images/dice6.jpg";
  }
}
</script>
