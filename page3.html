 function mouse_Click(event) {
  if(event.button==0)
  {
    document.getElementById("leftclick").innerHTML = "Mouse left click 'detected'";
  }
  if(event.button==1)
  {
    document.getElementById("scrollclick").innerHTML = "Scroll button click 'detected'";

 }
</script>

 <script>
   var resistor_1 = 49;
   var resistor_2 = 59;
   var resistor_3 = 69;
   var series_total = resistor_1 + resistor_2 + resistor_3;
   var parallel = 1/((1/resistor_1) + (1/resistor_2) + (1/resistor_3));
 
 
 function mouse_left_click(event) {
   if(event.button==0 && event.shiftKey) 
   {
     document.getElementById("shift_left_click").innerHTML =
 "Calculation of parallel connection resistance when: " + "<br>" +
 "resistor1 value = " + resistor_1 + "Ω" + "<br>" + 
  "resistor2 value = " + resistor_2 + "Ω" + "<br>" +
   "resistor3 value = " + resistor_3 + "Ω" + "<br>" +
     "Total parallel resistance = " + parallel.toFixed(2) + "Ω" + "<br>";
   }
   else if(event.button==0)
   {
     document.getElementById("left_click").innerHTML = 
 "Calculation of series (total) resistance when: " + "<br>" +
 "resistor1 value = " + resistor_1 + "Ω" + "<br>" + 
  "resistor2 value = " + resistor_2 + "Ω" + "<br>" +
   "resistor3 value = " + resistor_3 + "Ω" + "<br>" +
    "Total series resistance = " + series_total + "Ω" + "<br>";
   } 
  }
 </script>
