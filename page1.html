Tehtävä 1

Lyhyesti, mitä koodaus tekee. 

  <p id="Rinnankytky"></p>
  <p id="R1"></p>
  <p id="R2"></p>
  <p id="R3"></p>
  <p id="Rkok"></p>
  
  <script> 
  var otsikko = "Rinnankytkentä:";
  var R1 = 100;
  var R2 = 50;
  var R3 = 220;
  var Rkok = 1/((1/R1) + (1/R2) + (1/R3));
  
function rinnanlasku() {
document.getElementById("Rinnankytky").innerHTML = otsikko;
document.getElementById("R1").innerHTML = "R1 = " + R1 + "Ω";
document.getElementById("R2").innerHTML = "R2 = " + R2 + "Ω";
document.getElementById("R3").innerHTML = "R3 = " + R3 + "Ω";
document.getElementById("Rkok").innerHTML = "Rkok = " + Rkok.toFixed(2) + "Ω";
  }
  rinnanlasku();
  </script>

Tehtävä 3B

Lyhyesti, mitä koodaus tekee. 

  <p id="table_numbers";></p>
  <p id="table_numbers_avg";></p>
  
  <script>
  var table = [
    ["5", "10", "30", "100"],
    ["12", "33", "50", "1"],
    ["91", "20", "9", "14"]
  ];
  
  var table_sum = 0;
  for (var i = 0; i < table.length; i++) {
    for (var j = 0; j < table[i].length; j++) {
      table_sum += parseInt(table[i][j]);
    }
  }
  var table_avg = table_sum / (table.length * table[0].length);
  
  var table_rows = "";
    for (var i = 0; i < table.length; i++) {
      table_rows += table[i].join(", ") + "<br>";
    }
  
  document.getElementById("table_numbers").innerHTML = 
  "Table numbers are: <br><br>" + table_rows;
  document.getElementById("table_numbers_avg").innerHTML = 
  "Table numbers avg is: " + table_avg.toFixed(2);
  </script>
