<p id="tiedot"></p>
<p id="ala"></p>
<p id="A"></p>
<p id="tila"></p>
<p id="V"></p>

<script>

  var tiedot = "Laske pallon A Pinta-ala ja V Tilavuus kun säde on 15.";

  var Pii = 3.14;
  var R = 15; // R=sade (radius)

function A(){
  var A = 4 * Pii * R * R ;    // A=Pallon Pinta-ala (surface area)
  return A;
}

function V(){
  var V = 4 * Pii * R * R * R / 3;    // V=Pallon Tilavuus (volume)
  return V;
}

 document.getElementById("tiedot").innerHTML = tiedot;
 document.getElementById("ala").innerHTML = "Pallon Pinta-ala on:";
 document.getElementById("A").innerHTML = A();
 document.getElementById("tila").innerHTML = "Pallon Tilavuus on:";
 document.getElementById("V").innerHTML = V();

</script>
